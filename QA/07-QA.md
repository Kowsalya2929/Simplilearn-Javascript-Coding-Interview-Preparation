<h3>Question : How to sort a number array ?</h3>

```js
// Example

const unsortedArray = [1,22,45,11,100,4,67,34]

const arr1 = unsortedArray.sort() // Incorrect 

console.log(arr1) // [ 1, 100, 11, 22, 34, 4, 45, 67 ]

const arr2 = unsortedArray.sort((a, b) => a - b) // Correct - ascending order

console.log(arr2) // [  1,  4, 11,  22, 34, 45, 67, 100 ]

const arr3 = unsortedArray.sort((a, b) => b - a) // Correct - descending order

console.log(arr3) // [ 100, 67, 45, 34, 22, 11,  4,  1 ]
```
